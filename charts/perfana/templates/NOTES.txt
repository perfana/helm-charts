1. Get the application URL by running these commands:
{{- if .Values.perfanaWeb.ingress.enabled }}
{{- range .Values.perfanaWeb.ingress.hosts }}
  http{{ if $.Values.perfanaWeb.ingress.tls }}s{{ end }}://{{ .host }}
{{- end }}
{{- else if contains "NodePort" .Values.perfanaWeb.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services perfana-web)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.perfanaWeb.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get svc -w perfana-web'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} perfana-web -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SERVICE_IP:{{ .Values.perfanaWeb.service.port }}
{{- else if contains "ClusterIP" .Values.perfanaWeb.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app=perfana-web,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://localhost:4001 to use your application"
  kubectl port-forward $POD_NAME 4001:{{ .Values.perfanaWeb.service.port }}
{{- end }}

2. Check the status of all pods:
  kubectl get pods --namespace {{ .Release.Namespace }} -l "release={{ .Release.Name }}"

3. View logs for a specific component:
  kubectl logs --namespace {{ .Release.Namespace }} -l "app=perfana-api,release={{ .Release.Name }}"

For more information, visit: https://docs.perfana.io
